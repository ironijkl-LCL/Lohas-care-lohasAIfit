function doGet(e) { try { // 🎯 您的專屬試算表 ID var SPREADSHEET_ID = '1IEturpYhoHSEUlCr_VBoY8wv84RqHBaZuAOktHUFRO4'; var ss = SpreadsheetApp.openById(SPREADSHEET_ID); // ========================================== // 1. 讀取 "API" 工作表的資料 // ========================================== var apiSheet = ss.getSheetByName("API"); var apiData = {}; if (apiSheet) { apiData.apiName = apiSheet.getRange("A1").getValue() ? apiSheet.getRange("A1").getValue().toString().trim() : ""; apiData.apiKey = apiSheet.getRange("A2").getValue() ? apiSheet.getRange("A2").getValue().toString().trim() : ""; } else { apiData.error = "找不到名為 'API' 的工作表"; } // ========================================== // 2. 讀取 "nutrition_database" 工作表資料 // ========================================== var nutritionSheet = ss.getSheetByName("nutrition_database"); var foodList = []; if (nutritionSheet) { var lastRow = nutritionSheet.getLastRow(); var lastColumn = 18; // A 欄 到 R 欄共 18 欄 if (lastRow > 1) { // 取得第一行的欄位標頭 (A1:R1) var headers = nutritionSheet.getRange(1, 1, 1, lastColumn).getValues()[0]; // 取得第二行之後的所有食物數據 var dataRows = nutritionSheet.getRange(2, 1, lastRow - 1, lastColumn).getValues(); // 👥 檢查前端是否有傳入模糊搜尋關鍵字 (?query=xxx) var searchQuery = (e && e.parameter && e.parameter.query) ? e.parameter.query.toString().toLowerCase().trim() : ""; // 將數據組裝成 JSON 物件陣列 for (var i = 0; i < dataRows.length; i++) { var row = dataRows[i]; var foodNameInDb = row[0] ? row[0].toString() : ""; // A欄是 Food Name // 🎯【關鍵核心優化】：如果前端有傳 query，後端直接進行模糊匹配包含篩選 if (searchQuery !== "") { if (foodNameInDb.toLowerCase().indexOf(searchQuery) === -1) { continue; // 名字不包含關鍵字，直接跳過這筆 } } var foodItem = {}; for (var j = 0; j < headers.length; j++) { var headerName = headers[j] ? headers[j].toString().trim() : "Column_" + (j + 1); foodItem[headerName] = row[j]; } foodList.push(foodItem); } } } else { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "找不到名為 'nutrition_database' 的工作表，請檢查名稱" })).setMimeType(ContentService.MimeType.JSON); } // ========================================== // 3. 整合兩張工作表並回傳安全 JSON // ========================================== var responseData = { status: "success", api_config: apiData, nutrition_data: foodList }; return ContentService.createTextOutput(JSON.stringify(responseData)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "系統錯誤: " + error.toString() })).setMimeType(ContentService.MimeType.JSON); } }
